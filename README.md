- def calcular_media(lista):
    if len(lista) == 0:
        return 0
    total = sum(lista)
    media = total / len(lista)
    return media

# Exemplo de uso:
numeros = [4, 7, 9, 2, 5]
media = calcular_media(numeros)
print("A média dos números é:", media)
